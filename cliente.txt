// src/pages/admin/ClientesPage.jsx
import React, { useState, useMemo, useEffect } from 'react';
import AdminLayoutClean from './AdminLayoutClean';
import { initialCustomers } from '../../data';
import EntityDetailsModal from '../../components/EntityDetailsModal';
import EntityTable from '../../components/EntityTable';
import Alert from '../../components/Alert';

// Componente BlurModalOverlay
const BlurModalOverlay = ({ onClose, children }) => {
  return (
    <div
      style={{
        position: 'fixed',
        top: 0,
        left: 0,
        right: 0,
        bottom: 0,
        backgroundColor: 'rgba(0, 0, 0, 0.7)',
        backdropFilter: 'blur(8px)',
        WebkitBackdropFilter: 'blur(8px)',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        zIndex: 1000,
        padding: '20px',
      }}
      onClick={onClose}
    >
      <div
        onClick={(e) => e.stopPropagation()}
        style={{
          position: 'relative',
        }}
      >
        {children}
      </div>
    </div>
  );
};

const ClientesPage = () => {
  // ‚úÖ Estados principales
  const [clientes, setClientes] = useState([]);
  const [searchTerm, setSearchTerm] = useState('');
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [editingClient, setEditingClient] = useState(null);
  const [isDetailsOpen, setIsDetailsOpen] = useState(false);
  const [selectedClient, setSelectedClient] = useState(null);
  const [filterStatus, setFilterStatus] = useState('Todos');
  const [isDropdownOpen, setIsDropdownOpen] = useState(false);
  const [currentPage, setCurrentPage] = useState(1);
  const [itemsPerPage] = useState(8);
  const [alert, setAlert] = useState({ show: false, message: '', type: 'success' });
  
  // ‚úÖ Estados para departamentos y municipios
  const [departamentos, setDepartamentos] = useState([]);
  const [municipios, setMunicipios] = useState([]);
  const [isMunicipiosLoading, setIsMunicipiosLoading] = useState(false);
  
  // ‚úÖ Estado para errores de validaci√≥n
  const [errors, setErrors] = useState({});

  // ‚úÖ Funci√≥n para mostrar alertas mejorada
  const showAlert = (message, type = 'success') => {
    setAlert({ show: true, message, type });
    setTimeout(() => setAlert({ show: false, message: '', type: 'success' }), 3000);
  };

  // ‚úÖ Cargar departamentos de Colombia
  useEffect(() => {
    const fetchDepartamentos = async () => {
      try {
        const response = await fetch('https://api-colombia.com/api/v1/Department');
        const data = await response.json();
        setDepartamentos(data.sort((a, b) => a.name.localeCompare(b.name)));
      } catch (error) {
        console.error('Error cargando departamentos:', error);
        showAlert('Error cargando departamentos', 'delete');
      }
    };

    fetchDepartamentos();
  }, []);

  // ‚úÖ Cargar municipios cuando se selecciona un departamento
  const fetchMunicipiosByDepartamento = async (departamentoId) => {
    if (!departamentoId) {
      setMunicipios([]);
      return;
    }

    setIsMunicipiosLoading(true);
    try {
      const response = await fetch(`https://api-colombia.com/api/v1/Department/${departamentoId}/cities`);
      const data = await response.json();
      setMunicipios(data.sort((a, b) => a.name.localeCompare(b.name)));
    } catch (error) {
      console.error('Error cargando municipios:', error);
      showAlert('Error cargando municipios', 'delete');
      setMunicipios([]);
    } finally {
      setIsMunicipiosLoading(false);
    }
  };

  // ‚úÖ Carga e inicializaci√≥n de datos
  useEffect(() => {
    const mappedClientes = initialCustomers.map((cliente, index) => {
      return {
        id: cliente.Correo || `cliente-${index}`,
        tipoDocumento: 'C√©dula De Identidad',
        numeroDocumento: `DOC-${index + 1000}`,
        nombreCompleto: cliente.Nombre,
        email: cliente.Correo,
        telefono: cliente.Telefono,
        direccion: `${cliente.Direccion}, ${cliente.Ciudad}, ${cliente.Departamento}`,
        ciudad: cliente.Ciudad,
        departamento: cliente.Departamento,
        saldoFavor: cliente.SaldoaFavor,
        isActive: true,
      };
    });

    setClientes(mappedClientes);
  }, []);

  // ‚úÖ Filtro eficiente con useMemo
  const filteredClientes = useMemo(() => {
    const term = searchTerm.toLowerCase().trim();
    
    let filtered = clientes;
    
    if (filterStatus === 'Activos') {
      filtered = filtered.filter(cliente => cliente.isActive);
    } else if (filterStatus === 'Inactivos') {
      filtered = filtered.filter(cliente => !cliente.isActive);
    }
    
    if (term) {
      filtered = filtered.filter(cliente => {
        const nombreCompleto = cliente.nombreCompleto.toLowerCase();
        const email = cliente.email.toLowerCase();
        const documento = cliente.numeroDocumento.toLowerCase();
        const telefono = cliente.telefono.toLowerCase();

        return (
          nombreCompleto.includes(term) ||
          email.includes(term) ||
          documento.includes(term) ||
          telefono.includes(term)
        );
      });
    }
    
    return filtered;
  }, [clientes, searchTerm, filterStatus]);

  // ‚úÖ Paginaci√≥n
  const totalPages = Math.ceil(filteredClientes.length / itemsPerPage);
  const startIndex = (currentPage - 1) * itemsPerPage;
  const paginatedClientes = filteredClientes.slice(startIndex, startIndex + itemsPerPage);

  // ‚úÖ Columnas para EntityTable - CON SALDO A FAVOR
  const columns = [
    {
      header: 'Tipo',
      field: 'tipoDocumento',
      render: (item) => (
        <span style={{ color: '#fff', fontSize: '14px', fontWeight: '500' }}>
          {item.tipoDocumento}
        </span>
      )
    },
    {
      header: 'Documento',
      field: 'numeroDocumento',
      render: (item) => (
        <span style={{ color: '#fff', fontSize: '14px', fontWeight: '500' }}>
          {item.numeroDocumento}
        </span>
      )
    },
    {
      header: 'Nombre Completo',
      field: 'nombreCompleto',
      render: (item) => (
        <span style={{ color: '#fff', fontSize: '14px', fontWeight: '500' }}>
          {item.nombreCompleto}
        </span>
      )
    },
    {
      header: 'Email',
      field: 'email',
      render: (item) => (
        <span style={{ color: '#fff', fontSize: '14px' }}>
          {item.email}
        </span>
      )
    },
    {
      header: 'Tel√©fono',
      field: 'telefono',
      render: (item) => (
        <span style={{ color: '#fff', fontSize: '14px' }}>
          {item.telefono}
        </span>
      )
    },
    {
      header: 'Saldo A Favor',
      field: 'saldoFavor',
      render: (item) => (
        <span style={{ 
          fontSize: '14px',
          fontWeight: '600',
          color: item.saldoFavor !== '0' ? '#10b981' : '#9ca3af'
        }}>
          ${parseInt(item.saldoFavor || '0').toLocaleString()}
        </span>
      )
    }
  ];

  // ‚úÖ Validaci√≥n de campos
  const validateField = (name, value) => {
    const newErrors = { ...errors };
    
    switch (name) {
      case 'tipoDocumento':
        if (!value.trim()) {
          newErrors.tipoDocumento = 'Complete Este Campo';
        } else {
          delete newErrors.tipoDocumento;
        }
        break;
      case 'numeroDocumento':
        if (!value.trim()) {
          newErrors.numeroDocumento = 'Complete Este Campo';
        } else {
          delete newErrors.numeroDocumento;
        }
        break;
      case 'nombre':
        if (!value.trim()) {
          newErrors.nombre = 'Complete Este Campo';
        } else {
          delete newErrors.nombre;
        }
        break;
      case 'email':
        if (!value.trim()) {
          newErrors.email = 'Complete Este Campo';
        } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
          newErrors.email = 'Email Inv√°lido';
        } else {
          delete newErrors.email;
        }
        break;
      case 'telefono':
        if (!value.trim()) {
          newErrors.telefono = 'Complete Este Campo';
        } else {
          delete newErrors.telefono;
        }
        break;
      case 'direccion':
        if (!value.trim()) {
          newErrors.direccion = 'Complete Este Campo';
        } else {
          delete newErrors.direccion;
        }
        break;
      case 'ciudad':
        if (!value.trim()) {
          newErrors.ciudad = 'Complete Este Campo';
        } else {
          delete newErrors.ciudad;
        }
        break;
      case 'departamento':
        if (!value.trim()) {
          newErrors.departamento = 'Complete Este Campo';
        } else {
          delete newErrors.departamento;
        }
        break;
      default:
        break;
    }
    
    setErrors(newErrors);
  };

  // ‚úÖ Handlers de UI
  const openModal = (client = null) => {
    const defaultClient = {
      tipoDocumento: 'C√©dula De Identidad',
      numeroDocumento: '',
      nombre: '',
      email: '',
      telefono: '',
      direccion: '',
      ciudad: '',
      departamento: '',
      saldoFavor: '0',
      isActive: true,
    };

    if (client) {
      client = {
        ...client,
        nombre: client.nombreCompleto || ''
      };
    }

    setEditingClient(client ? { ...client } : defaultClient);
    setErrors({});
    setMunicipios([]);
    setIsModalOpen(true);
  };

  const closeModal = () => {
    setEditingClient(null);
    setErrors({});
    setMunicipios([]);
    setIsModalOpen(false);
  };

  const handleSave = (clientData) => {
    const { tipoDocumento, numeroDocumento, nombre, email, telefono, direccion, ciudad, departamento, saldoFavor } = clientData;
    
    // Validaci√≥n final antes de guardar
    const finalErrors = {};
    if (!tipoDocumento?.trim()) finalErrors.tipoDocumento = 'Complete Este Campo';
    if (!numeroDocumento?.trim()) finalErrors.numeroDocumento = 'Complete Este Campo';
    if (!nombre?.trim()) finalErrors.nombre = 'Complete Este Campo';
    if (!email?.trim()) finalErrors.email = 'Complete Este Campo';
    if (!telefono?.trim()) finalErrors.telefono = 'Complete Este Campo';
    if (!direccion?.trim()) finalErrors.direccion = 'Complete Este Campo';
    if (!ciudad?.trim()) finalErrors.ciudad = 'Complete Este Campo';
    if (!departamento?.trim()) finalErrors.departamento = 'Complete Este Campo';
    
    if (Object.keys(finalErrors).length > 0) {
      setErrors(finalErrors);
      return;
    }

    const updatedData = {
      ...clientData,
      tipoDocumento: tipoDocumento.trim(),
      numeroDocumento: numeroDocumento.trim(),
      nombreCompleto: nombre.trim(),
      nombre: nombre.trim(),
      email: email.trim(),
      telefono: telefono.trim(),
      direccion: direccion.trim(),
      ciudad: ciudad.trim(),
      departamento: departamento.trim(),
      saldoFavor: saldoFavor || '0',
      isActive: true,
    };

    if (updatedData.id) {
      setClientes(prev => prev.map(c => (c.id === updatedData.id ? updatedData : c)));
      showAlert(`Cliente "${updatedData.nombreCompleto}" actualizado correctamente`, 'edit');
    } else {
      const newId = updatedData.email || Date.now().toString();
      
      setClientes(prev => [
        ...prev,
        { 
          ...updatedData, 
          id: newId,
          tipoDocumento: updatedData.tipoDocumento,
          numeroDocumento: updatedData.numeroDocumento,
        }
      ]);
      showAlert(`Cliente "${updatedData.nombreCompleto}" creado correctamente`, 'add');
    }

    closeModal();
  };

  const handleToggleStatus = (id) => {
    const cliente = clientes.find(c => c.id === id);
    setClientes(prev =>
      prev.map(cliente =>
        cliente.id === id ? { ...cliente, isActive: !cliente.isActive } : cliente
      )
    );
    
    // Mostrar alerta de cambio de estado
    if (cliente) {
      const newStatus = !cliente.isActive;
      showAlert(
        `Cliente "${cliente.nombreCompleto}" ${newStatus ? 'activado' : 'desactivado'} correctamente`,
        newStatus ? 'add' : 'delete'
      );
    }
  };

  const viewClientDetails = (client) => {
    setSelectedClient(client);
    setIsDetailsOpen(true);
  };

  const handleFilterSelect = (status) => {
    setFilterStatus(status);
    setIsDropdownOpen(false);
    setCurrentPage(1);
  };

  // ‚úÖ Limpiar b√∫squeda
  const clearSearch = () => {
    setSearchTerm('');
    setCurrentPage(1);
  };

  // ‚úÖ Navegaci√≥n de p√°ginas
  const goToPage = (page) => {
    setCurrentPage(page);
  };

  const goToNextPage = () => {
    if (currentPage < totalPages) {
      setCurrentPage(currentPage + 1);
    }
  };

  const goToPrevPage = () => {
    if (currentPage > 1) {
      setCurrentPage(currentPage - 1);
    }
  };

  // ‚úÖ Handler para cambio de departamento
  const handleDepartamentoChange = (e) => {
    const departamentoId = e.target.value;
    const selectedDepartamento = departamentos.find(dep => dep.id.toString() === departamentoId);
    
    if (selectedDepartamento) {
      fetchMunicipiosByDepartamento(departamentoId);
    } else {
      setMunicipios([]);
    }
  };

  return (
    <AdminLayoutClean>
      {/* Alertas mejoradas */}
      {alert.show && (
        <Alert 
          message={alert.message} 
          type={alert.type} 
          onClose={() => setAlert({ show: false, message: '', type: 'success' })} 
          margin="30px"
        />
      )}

      {/* 1. üìã Encabezado */}
      <div style={{ 
        display: 'flex', 
        justifyContent: 'space-between', 
        alignItems: 'center', 
        marginBottom: '20px' 
      }}>
        <div>
          <h1 style={{ color: '#fff', fontSize: '1.8rem', fontWeight: '700' }}>Clientes</h1>
          <p style={{ color: '#bbb', fontSize: '13px' }}>Gestiona Los Clientes Registrados</p>
        </div>
        
        <button
          onClick={() => openModal()}
          style={{
            padding: '6px 12px',
            background: 'none',
            border: '1px solid #F5C81B',
            color: '#F5C81B',
            borderRadius: '4px',
            fontSize: '0.85rem',
            fontWeight: '600',
            display: 'flex',
            alignItems: 'center',
            gap: '4px',
          }}
        >
          Registrar Cliente
        </button>
      </div>

      {/* 2. üîç Barra de b√∫squeda + filtro */}
      <div style={{ 
        display: 'flex', 
        gap: '12px', 
        marginBottom: '20px',
        alignItems: 'center'
      }}>
        <div style={{ position: 'relative', flex: '1' }}>
          <input
            type="text"
            placeholder="Buscar Clientes Por Nombre, Email O Tel√©fono..."
            value={searchTerm}
            onChange={(e) => setSearchTerm(e.target.value)}
            style={{
              width: '100%',
              padding: '10px 16px 10px 40px',
              backgroundColor: '#1F2937',
              border: '1px solid #374151',
              color: '#fff',
              borderRadius: '8px',
              fontSize: '14px',
              outline: 'none',
              transition: 'border-color 0.2s',
              paddingRight: searchTerm ? '32px' : '16px',
            }}
            onFocus={(e) => e.target.style.borderColor = '#F5C81B'}
            onBlur={(e) => e.target.style.borderColor = '#374151'}
          />
          
          <div style={{
            position: 'absolute',
            left: '12px',
            top: '50%',
            transform: 'translateY(-50%)',
            color: '#9CA3AF',
          }}>
            <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
              <circle cx="11" cy="11" r="8"/>
              <path d="m21 21-4.3-4.3"/>
            </svg>
          </div>

          {searchTerm && (
            <button
              onClick={clearSearch}
              style={{
                position: 'absolute',
                right: '8px',
                top: '50%',
                transform: 'translateY(-50%)',
                background: 'none',
                border: 'none',
                color: '#9CA3AF',
                cursor: 'pointer',
                padding: '2px',
                borderRadius: '4px',
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                transition: 'all 0.2s',
                width: '16px',
                height: '16px'
              }}
              onMouseEnter={(e) => {
                e.currentTarget.style.color = '#fff';
                e.currentTarget.style.backgroundColor = 'rgba(255,255,255,0.1)';
              }}
              onMouseLeave={(e) => {
                e.currentTarget.style.color = '#9CA3AF';
                e.currentTarget.style.backgroundColor = 'transparent';
              }}
              title="Limpiar B√∫squeda"
            >
              <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                <line x1="18" y1="6" x2="6" y2="18"/>
                <line x1="6" y1="6" x2="18" y2="18"/>
              </svg>
            </button>
          )}
        </div>

        <div style={{ position: 'relative' }}>
          <button
            onClick={() => setIsDropdownOpen(!isDropdownOpen)}
            style={{
              display: 'flex',
              alignItems: 'center',
              gap: '8px',
              padding: '10px 16px',
              backgroundColor: 'transparent',
              border: '1px solid #F5C81B',
              color: '#F5C81B',
              borderRadius: '8px',
              fontSize: '14px',
              cursor: 'pointer',
              transition: 'all 0.2s',
              whiteSpace: 'nowrap',
              minWidth: '120px',
              justifyContent: 'space-between',
              fontWeight: '600'
            }}
            onMouseEnter={(e) => {
              e.target.style.backgroundColor = '#F5C81B';
              e.target.style.color = '#000';
            }}
            onMouseLeave={(e) => {
              e.target.style.backgroundColor = 'transparent';
              e.target.style.color = '#F5C81B';
            }}
          >
            <span>{filterStatus}</span>
            <svg 
              width="14" 
              height="14" 
              viewBox="0 0 24 24" 
              fill="none" 
              stroke="currentColor" 
              strokeWidth="2"
              style={{ 
                transform: isDropdownOpen ? 'rotate(180deg)' : 'rotate(0deg)',
                transition: 'transform 0.2s'
              }}
            >
              <polyline points="6,9 12,15 18,9"/>
            </svg>
          </button>

          {isDropdownOpen && (
            <div style={{
              position: 'absolute',
              top: '100%',
              right: 0,
              marginTop: '4px',
              backgroundColor: '#1F2937',
              border: '1px solid #F5C81B',
              borderRadius: '8px',
              padding: '8px 0',
              minWidth: '140px',
              zIndex: 1000,
              boxShadow: '0 4px 12px rgba(245, 200, 27, 0.3)'
            }}>
              {['Todos', 'Activos', 'Inactivos'].map((status) => (
                <button
                  key={status}
                  onClick={() => handleFilterSelect(status)}
                  style={{
                    width: '100%',
                    padding: '8px 16px',
                    backgroundColor: filterStatus === status ? '#F5C81B' : 'transparent',
                    border: 'none',
                    color: filterStatus === status ? '#000' : '#F5C81B',
                    fontSize: '14px',
                    textAlign: 'left',
                    cursor: 'pointer',
                    transition: 'all 0.2s',
                    fontWeight: filterStatus === status ? '600' : '400'
                  }}
                  onMouseEnter={(e) => {
                    if (filterStatus !== status) {
                      e.target.style.backgroundColor = '#F5C81B';
                      e.target.style.color = '#000';
                    }
                  }}
                  onMouseLeave={(e) => {
                    if (filterStatus !== status) {
                      e.target.style.backgroundColor = 'transparent';
                      e.target.style.color = '#F5C81B';
                    }
                  }}
                >
                  {status}
                </button>
              ))}
            </div>
          )}
        </div>
      </div>

      {/* 3. üìä Tabla de Clientes CON SALDO A FAVOR */}
      <EntityTable
        entities={paginatedClientes}
        columns={columns}
        onView={viewClientDetails}
        onEdit={openModal}
        onToggleStatus={handleToggleStatus}
        idField="id"
        isActiveField="isActive"
        getName={(item) => item.nombreCompleto}
      />

      {/* 4. üîÑ Paginaci√≥n */}
      {totalPages > 1 && (
        <div style={{
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center',
          padding: '16px 20px',
          backgroundColor: '#1F2937',
          borderTop: '1px solid #374151',
          borderRadius: '0 0 8px 8px'
        }}>
          <button
            onClick={goToPrevPage}
            disabled={currentPage === 1}
            style={{
              padding: '8px 16px',
              backgroundColor: 'transparent',
              border: '1px solid #F5C81B',
              color: '#F5C81B',
              borderRadius: '6px',
              fontSize: '14px',
              cursor: currentPage === 1 ? 'not-allowed' : 'pointer',
              opacity: currentPage === 1 ? 0.5 : 1,
              display: 'flex',
              alignItems: 'center',
              gap: '8px',
              transition: 'all 0.2s'
            }}
            onMouseEnter={(e) => {
              if (currentPage !== 1) {
                e.target.style.backgroundColor = '#F5C81B';
                e.target.style.color = '#000';
              }
            }}
            onMouseLeave={(e) => {
              if (currentPage !== 1) {
                e.target.style.backgroundColor = 'transparent';
                e.target.style.color = '#F5C81B';
              }
            }}
          >
            <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
              <polyline points="15,18 9,12 15,6"/>
            </svg>
            Anterior
          </button>

          <div style={{ display: 'flex', gap: '8px', alignItems: 'center' }}>
            {Array.from({ length: totalPages }, (_, i) => i + 1).map((page) => (
              <button
                key={page}
                onClick={() => goToPage(page)}
                style={{
                  width: '32px',
                  height: '32px',
                  borderRadius: '6px',
                  border: '1px solid',
                  borderColor: currentPage === page ? '#F5C81B' : '#374151',
                  backgroundColor: currentPage === page ? '#F5C81B' : 'transparent',
                  color: currentPage === page ? '#000' : '#9CA3AF',
                  fontSize: '14px',
                  fontWeight: '600',
                  cursor: 'pointer',
                  transition: 'all 0.2s'
                }}
                onMouseEnter={(e) => {
                  if (currentPage !== page) {
                    e.target.style.backgroundColor = '#374151';
                    e.target.style.color = '#F5C81B';
                  }
                }}
                onMouseLeave={(e) => {
                  if (currentPage !== page) {
                    e.target.style.backgroundColor = 'transparent';
                    e.target.style.color = '#9CA3AF';
                  }
                }}
              >
                {page}
              </button>
            ))}
          </div>

          <button
            onClick={goToNextPage}
            disabled={currentPage === totalPages}
            style={{
              padding: '8px 16px',
              backgroundColor: 'transparent',
              border: '1px solid #F5C81B',
              color: '#F5C81B',
              borderRadius: '6px',
              fontSize: '14px',
              cursor: currentPage === totalPages ? 'not-allowed' : 'pointer',
              opacity: currentPage === totalPages ? 0.5 : 1,
              display: 'flex',
              alignItems: 'center',
              gap: '8px',
              transition: 'all 0.2s'
            }}
            onMouseEnter={(e) => {
              if (currentPage !== totalPages) {
                e.target.style.backgroundColor = '#F5C81B';
                e.target.style.color = '#000';
              }
            }}
            onMouseLeave={(e) => {
              if (currentPage !== totalPages) {
                e.target.style.backgroundColor = 'transparent';

                    fontWeight: '600',
                    cursor: 'pointer',
                    transition: 'all 0.2s',
                    marginTop: '8px',
                  }}
                  onMouseEnter={(e) => {
                    e.currentTarget.style.backgroundColor = '#F5C81B';
                    e.currentTarget.style.color = '#000';
                  }}
                  onMouseLeave={(e) => {
                    e.currentTarget.style.backgroundColor = 'transparent';
                    e.currentTarget.style.color = '#F5C81B';
                  }}
                >
                  Cerrar
                </button>
              </div>
            )}
          </div>
        </BlurModalOverlay>
      )}
    </AdminLayoutClean>
  );
};
